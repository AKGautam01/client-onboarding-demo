  <script>
    fetch("http://localhost:1337/api/client-onboarding-requests")
      .then(res => res.json())
      .then(data => {
        const list = document.getElementById("entries");
        data.data.forEach(entry => {
          const item = document.createElement("li");
          item.textContent = `Contact Person: ${entry.ContactPersonRohanSharma}, Number: ${entry.ContactNumber9712485755}`;
          list.appendChild(item);
        });
      });
  </script>
